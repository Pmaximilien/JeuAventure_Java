001    /**
002     * This class is part of the "World of Zuul" application. 
003     * "World of Zuul" is a very simple, text based adventure game.  
004     * 
005     * This class holds an enumeration of all command words known to the game.
006     * It is used to recognise commands as they are typed in.
007     *
008     * @author  Michael Kolling and David J. Barnes
009     * @version 2006.03.30
010     */
011    
012    public class CommandWords
013    {
014        // a constant array that holds all valid command words
015        private static final String[] validCommands = {
016            "go", "quit", "help", "look", "eat", "status"
017        };
018    
019        /**
020         * Constructor - initialise the command words.
021         */
022        public CommandWords()
023        {
024            // nothing to do at the moment...
025        }
026    
027        /**
028         * Check whether a given String is a valid command word. 
029         * @return true if a given string is a valid command,
030         * false if it isn't.
031         */
032        public boolean isCommand(String aString)
033        {
034            for(int i = 0; i < validCommands.length; i++) {
035                if(validCommands[i].equals(aString))
036                    return true;
037            }
038            // if we get here, the string was not found in the commands
039            return false;
040        }
041    
042        public String getCommandList()
043        {
044            String ResCmd = new String();
045    
046            for(int i = 0; i < validCommands.length; i++)
047                    ResCmd += (validCommands[i] + " ");
048            return ResCmd;
049        }
050    }
