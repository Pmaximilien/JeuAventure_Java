005    public class Item

008            private int poids;
009            private String description;

012            public Item(String init_nom)
013            {
014                    nom_item = new String(init_nom);
015            }
016

018            {
019                    nom_item = new String(init_nom);
020                    prix = _prix;
021                    poids = _poids;
022            }
023    
024            public void setIDescription(String _description)
025            {
026                    description = new String(_description);
027            }
028    
029            public int getPrix()
030            {
031                    return prix;
032            }
033            
034            public int getPoids()
035            {
036                    return poids;
037            }
038            public String getInfoItem(){
039                    String resInfo = new String("nom ");
040    
041                    resInfo += nom_item + " " + poids + " " + prix;
042    
043                    return resInfo;
044            }
045    }
